cmake_minimum_required(VERSION 3.22)
project(Sofa.Component.IntegrationSchemes.Forward LANGUAGES CXX)

set(SOFACOMPONENTINTEGRATIONSCHEMESFORWARD_SOURCE_DIR "src/sofa/component/integrationschemes/forward")

set(HEADER_FILES
    ${SOFACOMPONENTINTEGRATIONSCHEMESFORWARD_SOURCE_DIR}/config.h.in
    ${SOFACOMPONENTINTEGRATIONSCHEMESFORWARD_SOURCE_DIR}/init.h
    ${SOFACOMPONENTINTEGRATIONSCHEMESFORWARD_SOURCE_DIR}/RungeKutta2Solver.h
    ${SOFACOMPONENTINTEGRATIONSCHEMESFORWARD_SOURCE_DIR}/RungeKutta4Solver.h
    ${SOFACOMPONENTINTEGRATIONSCHEMESFORWARD_SOURCE_DIR}/EulerExplicitSolver.h
    ${SOFACOMPONENTINTEGRATIONSCHEMESFORWARD_SOURCE_DIR}/DampVelocitySolver.h
    ${SOFACOMPONENTINTEGRATIONSCHEMESFORWARD_SOURCE_DIR}/CentralDifferenceSolver.h
)

set(SOURCE_FILES
    ${SOFACOMPONENTINTEGRATIONSCHEMESFORWARD_SOURCE_DIR}/init.cpp
    ${SOFACOMPONENTINTEGRATIONSCHEMESFORWARD_SOURCE_DIR}/RungeKutta2Solver.cpp
    ${SOFACOMPONENTINTEGRATIONSCHEMESFORWARD_SOURCE_DIR}/RungeKutta4Solver.cpp
    ${SOFACOMPONENTINTEGRATIONSCHEMESFORWARD_SOURCE_DIR}/EulerExplicitSolver.cpp
    ${SOFACOMPONENTINTEGRATIONSCHEMESFORWARD_SOURCE_DIR}/DampVelocitySolver.cpp
    ${SOFACOMPONENTINTEGRATIONSCHEMESFORWARD_SOURCE_DIR}/CentralDifferenceSolver.cpp
)

sofa_find_package(Sofa.Simulation.Core REQUIRED)

add_library(${PROJECT_NAME} SHARED ${HEADER_FILES} ${SOURCE_FILES})
target_link_libraries(${PROJECT_NAME} PUBLIC Sofa.Simulation.Core)

sofa_create_package_with_targets(
    PACKAGE_NAME ${PROJECT_NAME}
    PACKAGE_VERSION ${Sofa_VERSION}
    TARGETS ${PROJECT_NAME} AUTO_SET_TARGET_PROPERTIES
    INCLUDE_SOURCE_DIR "src"
    INCLUDE_INSTALL_DIR "${PROJECT_NAME}"
)
